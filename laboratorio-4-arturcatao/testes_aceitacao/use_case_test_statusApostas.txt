# Testes de Aceitacao - statusApostas()

## Cenario 1: Sem apostas registradas ##
expect "NAO TEM APOSTAS REGISTRADAS!" statusAposta

## Cenario 2: Com apostas registradas ##
# Set-up inicial
incluirTime codigo="250_PB" nome="Campinense" mascote="Raposa"
incluirCampeonato nome="Copa do Nordeste 2023" participantes=10
incluirTimeCampeonato codigo="250_PB" campeonato="Copa do Nordeste 2023"

# Registrando apostas
tentarSorte codTime="250_PB" nomeCamp="Copa do Nordeste 2023" colocacao=1 preco=100.0
tentarSorte codTime="250_PB" nomeCamp="Copa do Nordeste 2023" colocacao=5 preco=50.0

# Verificando status
expect "Apostas: | 1. [250_PB] Campinense / Raposa | Copa do Nordeste 2023 - 1/10 | R$100.0 | 2. [250_PB] Campinense / Raposa | Copa do Nordeste 2023 - 5/10 | R$50.0" statusAposta

## Cenario 3: Com multiplas apostas ##
# Adicionando mais uma aposta
incluirTime codigo="252_PB" nome="Treze" mascote="Galo"
incluirCampeonato nome="Brasileirao Serie A 2023" participantes=20
incluirTimeCampeonato codigo="252_PB" campeonato="Brasileirao Serie A 2023"
tentarSorte codTime="252_PB" nomeCamp="Brasileirao Serie A 2023" colocacao=3 preco=200.0

# Verificando status atualizado
expect "Apostas: | 1. [250_PB] Campinense / Raposa | Copa do Nordeste 2023 - 1/10 | R$100.0 | 2. [250_PB] Campinense / Raposa | Copa do Nordeste 2023 - 5/10 | R$50.0 | 3. [252_PB] Treze / Galo | Brasileirao Serie A 2023 - 3/20 | R$200.0" statusAposta

## Cenario 4: Ordenacao das apostas ##
# Verificando se mantem a ordem de inserção
tentarSorte codTime="250_PB" nomeCamp="Copa do Nordeste 2023" colocacao=2 preco=75.0
expect "Apostas: | 1. [250_PB] Campinense / Raposa | Copa do Nordeste 2023 - 1/10 | R$100.0 | 2. [250_PB] Campinense / Raposa | Copa do Nordeste 2023 - 5/10 | R$50.0 | 3. [252_PB] Treze / Galo | Brasileirao Serie A 2023 - 3/20 | R$200.0 | 4. [250_PB] Campinense / Raposa | Copa do Nordeste 2023 - 2/10 | R$75.0" statusAposta
